def xor(c: str, n: int) -> str:
    temp = ord(c)
    for _ in range(n):
        temp ^= n
    return chr(temp)

alp = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ_!{}"
c = "nhtjcZcsfroydRx`rl"

m = ""
for i in range(len(c)):
    target = c[i]
    for a in alp:
        ch = xor(a, i)
        if ch == target:
            m += a
            break

print(m)
