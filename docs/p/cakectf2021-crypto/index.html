with open("output.txt") as f:
	buf = f.read().split("\n")

n = eval(buf[0].split()[2])

y = eval(buf[3].split()[2])

pr = n // q

r = gcd(y - x + q, pr)

p = pr // r

d = inverse_mod(e, phi)
